638 static void controlAction(ControlLoopVariables& motor) {
639  //Calculate Error
640  motor.motor_error = motor.motor_setpoint - motor.motor_input;
641  //calculate Intgral and Derivative terms for each motor
642  motor.motor_integral += motor.motor_error * elapsed_time;
643  motor.motor_integral = constrain(motor.motor_integral, -1 * motor.motorGains->iSat, motor.motorGains->iSat);
644  motor.motor_derivative = (motor.motor_error - motor.motor_prev_error) / elapsed_time;
645  motor.motor_prev_error = motor.motor_error;
