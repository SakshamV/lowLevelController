715 static void controlAction(ControlLoopVariables& motor) {
716  //Calculate Error
717  motor.motor_error = motor.motor_setpoint - motor.motor_input;
718  //calculate Intgral and Derivative terms for each motor
719  motor.motor_integral += motor.motor_error * elapsed_time;
720  motor.motor_integral = constrain(motor.motor_integral, -1 * motor.motorGains->iSat, motor.motorGains->iSat);
721  motor.motor_derivative = (motor.motor_error - motor.motor_prev_error) / elapsed_time;
722  motor.motor_prev_error = motor.motor_error;
723  //Calculate Outut
724  motor.motor_output = motor.motorGains->kp * motor.motor_error + motor.motorGains->ki * motor.motor_integral + motor.motorGains->kd * motor.motor_derivative;
725 }
